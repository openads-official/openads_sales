데이터 관리 · 동기화

구글 시트를 비공개로 유지하면서 대시보드와 연결하는 방법

⚙️ 방법 C · Apps Script 웹앱 배포 (권장)

구글 시트 자체를 공개하는 게 아니라, Apps Script만 내부 전용 API로 배포하는 방식입니다. 시트 접근 권한은 그대로 유지하면서 대시보드만 데이터를 읽어올 수 있어요.

✅ 시트 비공개 유지
구글 시트 공유 설정 변경 없음. 기존 팀 권한 그대로.
✅ 조직 내부 접근만 허용
같은 Google Workspace 계정만 API에 접근 가능.
✅ 매주 자동 동기화
트리거 설정 시 매주 월요일 자동으로 데이터 갱신.
설정 순서 (최초 1회만)
  • 1
    Apps Script 열기
    구글 시트 상단 메뉴 → 확장 프로그램 → Apps Script 클릭
  • 2
    아래 스크립트 전체 복사 후 붙여넣기
    기존 코드 전체 삭제 후 아래 코드 붙여넣기 → 저장(Ctrl+S)
  • 3
    웹앱으로 배포
    Apps Script 우측 상단 배포 → 새 배포 → 유형: 웹 앱 선택
    ⚠️ 실행 계정: 나 · 액세스 권한: 조직 내부 Google Workspace 사용자
  • 4
    배포 URL 복사
    배포 완료 후 생성된 웹앱 URL을 복사해 아래 입력란에 붙여넣기
  • 5
    매주 자동 트리거 설정
    Apps Script 좌측 메뉴 ⏰ 트리거 → 트리거 추가 → setWeeklyTrigger 함수 → 시간 기반 → 매주 월요일 오전 9시
📋 Apps Script 코드 (전체 복사 후 붙여넣기)
// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ // 오픈애즈 세일즈 인텔리전스 · Apps Script 웹앱 // 구글 시트 → 비공개 내부 API 변환 // ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ // ✅ GET 요청 처리: 대시보드에서 데이터 요청 시 응답 function doGet(e) { const sheet = e.parameter.sheet || 'revenue'; const data = getData(sheet); // CORS 헤더 포함하여 JSON 반환 return ContentService .createTextOutput(JSON.stringify({ ok: true, data: data, updatedAt: new Date().toISOString() })) .setMimeType(ContentService.MimeType.JSON); } // 시트별 데이터 추출 function getData(sheetName) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheetMap = { 'revenue' : '26_광고매출현황_r', 'pipeline' : '26_세일즈파이프라인', 'efficiency': '광고효율' }; const targetSheet = ss.getSheetByName(sheetMap[sheetName] || sheetMap['revenue']); if (!targetSheet) return []; const raw = targetSheet.getDataRange().getValues(); // ★ 매출: 26_광고매출현황_r 시트 행5~16 (월별 집계행) // 컬럼: A=연도, B=월, C=인바운드, D=아웃바운드, E=전체, F=드롭, G=집행건수, H=매출(만원) if (sheetName === 'revenue') { const monthly = []; for (let i = 4; i <= 16; i++) { const row = raw[i]; if (!row || !row[0] || !row[1]) continue; const yr = Number(row[0]); const mo = Number(row[1]); const rev = Number(row[7]); // H열: 매출(만원) — 집행월 기준 SUMIFS 수식 결과 if (!yr || !mo) continue; monthly.push({ year : yr, month : mo, inbound : Number(row[2]) || 0, outbound: Number(row[3]) || 0, total : Number(row[4]) || 0, dropped : Number(row[5]) || 0, executed: Number(row[6]) || 0, revenue : isNaN(rev) ? 0 : rev, // ★ 0원도 명시적으로 반환 target25: Number(row[8]) || 0, yoy : Number(row[9]) || 0 }); } return monthly; } // 파이프라인 (헤더 1행 + 데이터) if (sheetName === 'pipeline') { const header = raw[0].map(h => String(h).trim()); const rows = []; for (let i = 1; i < raw.length; i++) { const row = raw[i]; // 제안 날짜(G열, index 6) 없으면 skip if (!row[6] && !row[10]) continue; const obj = {}; header.forEach((h, j) => { // 날짜 객체는 ISO 문자열로 변환 obj[h] = (row[j] instanceof Date) ? row[j].toISOString() : row[j]; }); rows.push(obj); } return rows; } // 광고효율 — 헤더 1행, 데이터 4행(index 3)부터 // 컬럼: 광고기간 / 상품 유형 / 광고주 / 대행사 / 소재 / 컨텐츠(제목) / // 매출(만원) / 발송수 / 오픈수 / 오픈율 / 오픈수(중복포함) / 오픈율(중복포함) / // 클릭수 / CTR / 수신거부 if (sheetName === 'efficiency') { const header = raw[0].map(h => String(h).trim()); const rows = []; for (let i = 3; i < raw.length; i++) { const row = raw[i]; // 광고주(3번째 컬럼, index 2)가 없으면 빈 행으로 skip if (!row[2] || String(row[2]).trim() === '') continue; const obj = {}; header.forEach((h, j) => { // Date 객체는 ISO 문자열로 변환, 숫자는 그대로 if (row[j] instanceof Date) { obj[h] = row[j].toISOString(); } else { obj[h] = row[j]; } }); rows.push(obj); } return rows; } return raw; } // ⏰ 매주 월요일 오전 9시 자동 트리거 설정 (최초 1회 실행) function setWeeklyTrigger() { // 기존 트리거 모두 삭제 후 재등록 ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t)); ScriptApp.newTrigger('doGet') .timeBased() .onWeekDay(ScriptApp.WeekDay.MONDAY) .atHour(9) .create(); Logger.log('✅ 트리거 설정 완료: 매주 월요일 오전 9시'); } // 🔍 테스트용: 로그로 데이터 확인 function testData() { Logger.log(JSON.stringify(getData('revenue'), null, 2)); Logger.log(JSON.stringify(getData('pipeline').slice(0, 3), null, 2)); }
