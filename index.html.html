데이터 관리 · 동기화

구글 시트를 비공개로 유지하면서 대시보드와 연결하는 방법

⚙️ 방법 C · Apps Script 웹앱 배포 (권장)

구글 시트 자체를 공개하는 게 아니라, Apps Script만 내부 전용 API로 배포하는 방식입니다. 시트 접근 권한은 그대로 유지하면서 대시보드만 데이터를 읽어올 수 있어요.

✅ 시트 비공개 유지
구글 시트 공유 설정 변경 없음. 기존 팀 권한 그대로.
✅ 조직 내부 접근만 허용
같은 Google Workspace 계정만 API에 접근 가능.
✅ 매주 자동 동기화
트리거 설정 시 매주 월요일 자동으로 데이터 갱신.
설정 순서 (최초 1회만)
  • 1
    Apps Script 열기
    구글 시트 상단 메뉴 → 확장 프로그램 → Apps Script 클릭
  • 2
    아래 스크립트 전체 복사 후 붙여넣기
    기존 코드 전체 삭제 후 아래 코드 붙여넣기 → 저장(Ctrl+S)
  • 3
    웹앱으로 배포
    Apps Script 우측 상단 배포 → 새 배포 → 유형: 웹 앱 선택
    ⚠️ 실행 계정: 나 · 액세스 권한: 조직 내부 Google Workspace 사용자
  • 4
    배포 URL 복사
    배포 완료 후 생성된 웹앱 URL을 복사해 아래 입력란에 붙여넣기
  • 5
    매주 자동 트리거 설정
    Apps Script 좌측 메뉴 ⏰ 트리거 → 트리거 추가 → setWeeklyTrigger 함수 → 시간 기반 → 매주 월요일 오전 9시
📋 Apps Script 코드 (전체 복사 후 붙여넣기)
// ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ // 오픈애즈 세일즈 인텔리전스 · Apps Script 웹앱 // 구글 시트 → 비공개 내부 API 변환 // ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━ // ✅ GET 요청 처리: 대시보드에서 데이터 요청 시 응답 function doGet(e) { const sheet = e.parameter.sheet || 'revenue'; const data = getData(sheet); // CORS 헤더 포함하여 JSON 반환 return ContentService .createTextOutput(JSON.stringify({ ok: true, data: data, updatedAt: new Date().toISOString() })) .setMimeType(ContentService.MimeType.JSON); } // 시트별 데이터 추출 function getData(sheetName) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheetMap = { 'revenue' : '26_광고매출현황_r', 'pipeline' : '26_세일즈파이프라인', 'efficiency': '광고효율' }; const targetSheet = ss.getSheetByName(sheetMap[sheetName] || sheetMap['revenue']); if (!targetSheet) return []; const raw = targetSheet.getDataRange().getValues(); // 월별 매출 요약 (revenue 시트 기준: 행5~16 = 1~12월) if (sheetName === 'revenue') { const monthly = []; for (let i = 4; i <= 15; i++) { const row = raw[i]; if (!row || !row[1]) continue; monthly.push({ year : row[0], month : row[1], inbound : row[2] || 0, outbound : row[3] || 0, total : row[4] || 0, dropped : row[5] || 0, executed : row[6] || 0, revenue : row[7] || 0, target25 : row[8] || 0, yoy : row[9] || 0 }); } return monthly; } // 파이프라인 (pipeline 시트: 헤더 1행 + 데이터) if (sheetName === 'pipeline') { const header = raw[0]; const rows = []; for (let i = 1; i < raw.length; i++) { const row = raw[i]; if (!row[6]) continue; // 제안 날짜 없으면 skip const obj = {}; header.forEach((h, j) => { obj[h] = row[j]; }); rows.push(obj); } return rows; } // 광고효율 (efficiency 시트) if (sheetName === 'efficiency') { const header = raw[0]; const rows = []; for (let i = 3; i < raw.length; i++) { const row = raw[i]; if (!row[0]) continue; const obj = {}; header.forEach((h, j) => { obj[h] = row[j]; }); rows.push(obj); } return rows; } return raw; } // ⏰ 매주 월요일 오전 9시 자동 트리거 설정 (최초 1회 실행) function setWeeklyTrigger() { // 기존 트리거 모두 삭제 후 재등록 ScriptApp.getProjectTriggers().forEach(t => ScriptApp.deleteTrigger(t)); ScriptApp.newTrigger('doGet') .timeBased() .onWeekDay(ScriptApp.WeekDay.MONDAY) .atHour(9) .create(); Logger.log('✅ 트리거 설정 완료: 매주 월요일 오전 9시'); } // 🔍 테스트용: 로그로 데이터 확인 function testData() { Logger.log(JSON.stringify(getData('revenue'), null, 2)); Logger.log(JSON.stringify(getData('pipeline').slice(0, 3), null, 2)); }
